#include <Arduino.h>
#include <FastLED.h>
#include "Joystick.h"

#define LED_PIN_OBEN   26  
#define LED_PIN_UNTEN  25  
#define COLOR_ORDER    GRB
#define CHIPSET        WS2812B
#define WIDTH  32
#define HEIGHT_TOTAL 16

const int PIN_X = 34;
const int PIN_Y = 35;
const int PIN_BTN = 32;

CRGB ledsOben[256];
CRGB ledsUnten[256];
Joystick joystick(PIN_X, PIN_Y, PIN_BTN);

struct Point { int x, y; };
Point snake[3]; 
Point dir = {1, 0}; // Standardmäßig nach rechts starten
unsigned long lastMoveTime = 0;
const int moveInterval = 200; 

// --- Mapping Funktion bleibt gleich ---
void setPixel(int x, int y, CRGB color) {
  if (y < 0 || y >= 16 || x < 0 || x >= 32) return;
  if (y < 8) { 
    int index;
    if (x % 2 == 0) index = x * 8 + y;
    else index = x * 8 + (7 - y);
    if (index >= 0 && index < 256) ledsOben[index] = color;
  } 
  else { 
    int vX = 31 - x;
    int vY = 7 - (y - 8);
    int index;
    if (vX % 2 == 0) index = vX * 8 + vY;
    else index = vX * 8 + (7 - vY);
    if (index >= 0 && index < 256) ledsUnten[index] = color;
  }
}

void resetSnake() {
  snake[0] = {15, 8};
  snake[1] = {14, 8};
  snake[2] = {13, 8};
  dir = {1, 0}; // Startrichtung nach Reset
}

void setup() {
  Serial.begin(115200);
  FastLED.addLeds<CHIPSET, LED_PIN_OBEN, COLOR_ORDER>(ledsOben, 256);
  FastLED.addLeds<CHIPSET, LED_PIN_UNTEN, COLOR_ORDER>(ledsUnten, 256);
  FastLED.setBrightness(15);
  resetSnake();
}

void loop() {
  // 1. STEUERUNG: Nur ändern, wenn Schwellenwert erreicht wird
  int xVal = analogRead(PIN_X);
  int yVal = analogRead(PIN_Y);

  // WICHTIG: Die Abfrage "&& dir.x == 0" verhindert das Umkehren in sich selbst
  if (xVal < 500) { 
      if (dir.x == 0) { dir = {-1, 0}; } // Links
  }
  else if (xVal > 3500) { 
      if (dir.x == 0) { dir = {1, 0}; }  // Rechts
  }
  else if (yVal < 500) { 
      if (dir.y == 0) { dir = {0, -1}; } // Oben
  }
  else if (yVal > 3500) { 
      if (dir.y == 0) { dir = {0, 1}; }  // Unten
  }

  // 2. BEWEGUNG
  if (millis() - lastMoveTime > moveInterval) {
    lastMoveTime = millis();

    int nextX = snake[0].x + dir.x;
    int nextY = snake[0].y + dir.y;

    // Kollision Wand
    if (nextX < 0 || nextX >= WIDTH || nextY < 0 || nextY >= HEIGHT_TOTAL) {
      for(int i=0; i<3; i++) setPixel(snake[i].x, snake[i].y, CRGB::Red);
      FastLED.show();
      delay(800);
      resetSnake();
      return;
    }

    // Körper folgt Kopf
    for (int i = 2; i > 0; i--) snake[i] = snake[i - 1];
    snake[0].x = nextX;
    snake[0].y = nextY;

    // 3. ZEICHNEN
    FastLED.clear();
    
    // Rahmen zeichnen
    for(int x=0; x<WIDTH; x++) { setPixel(x, 0, CRGB(5, 5, 20)); setPixel(x, 15, CRGB(5, 5, 20)); }
    for(int y=0; y<HEIGHT_TOTAL; y++) { setPixel(0, y, CRGB(5, 5, 20)); setPixel(31, y, CRGB(5, 5, 20)); }

    // Schlange zeichnen
    for (int i = 0; i < 3; i++) {
      setPixel(snake[i].x, snake[i].y, (i == 0) ? CRGB::White : CRGB::Green);
    }
    FastLED.show();
  }
}
