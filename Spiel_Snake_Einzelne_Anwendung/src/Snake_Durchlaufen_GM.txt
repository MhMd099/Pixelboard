#include <Arduino.h>
#include <FastLED.h>

#define LED_PIN_OBEN   26  
#define LED_PIN_UNTEN  25  
#define COLOR_ORDER    GRB
#define CHIPSET        WS2812B

// Wir behandeln beide Paneele erst einmal als zwei lange Ketten hintereinander
CRGB ledsOben[256];
CRGB ledsUnten[256];

void setup() {
  Serial.begin(115200);
  FastLED.addLeds<CHIPSET, LED_PIN_OBEN, COLOR_ORDER>(ledsOben, 256);
  FastLED.addLeds<CHIPSET, LED_PIN_UNTEN, COLOR_ORDER>(ledsUnten, 256);
  FastLED.setBrightness(10);
  FastLED.clear();
}

void loop() {
  // --- TEST 1: Oberes Paneel (Pin 26) ---
  // Wir laufen jede LED einzeln durch von 0 bis 255
  for (int i = 0; i < 256; i++) {
    FastLED.clear();
    
    // Aktuelle LED weiß leuchten lassen
    ledsOben[i] = CRGB::White;
    
    // Ein paar "Folge-Pixel" (die Schlange), damit man die Richtung sieht
    if (i > 0) ledsOben[i-1] = CRGB::Green;
    if (i > 1) ledsOben[i-2] = CRGB::Green;

    Serial.printf("Paneel OBEN - LED Index: %d\n", i);
    FastLED.show();
    delay(50); // Geschwindigkeit
  }

  // --- TEST 2: Unteres Paneel (Pin 25) ---
  for (int i = 0; i < 256; i++) {
    FastLED.clear();
    
    ledsUnten[i] = CRGB::White;
    
    if (i > 0) ledsUnten[i-1] = CRGB::Blue;
    if (i > 1) ledsUnten[i-2] = CRGB::Blue;

    Serial.printf("Paneel UNTEN - LED Index: %d\n", i);
    FastLED.show();
    delay(50);
  }
}
