#include <Arduino.h>
#include <FastLED.h>
#include "Joystick.h"

#define LED_PIN_OBEN   26  
#define LED_PIN_UNTEN  25  
#define COLOR_ORDER    GRB
#define CHIPSET        WS2812B
#define WIDTH  32
#define HEIGHT_TOTAL 16

// Hardware-Pins für den Joystick
const int PIN_X = 34;
const int PIN_Y = 35;
const int PIN_BTN = 32;

CRGB ledsOben[256];
CRGB ledsUnten[256];

Joystick joystick(PIN_X, PIN_Y, PIN_BTN);

// --- Snake Variablen ---
struct Point { int x, y; };
Point snake[3]; // Wir bleiben erst mal bei 3 Punkten
Point dir = {1, 0}; // Startrichtung: nach rechts
unsigned long lastMoveTime = 0;
const int moveInterval = 150; // Langsame Geschwindigkeit

// Deine bewährte Mapping-Funktion
void setPixel(int x, int y, CRGB color) {
  if (y < 0 || y >= 16 || x < 0 || x >= 32) return;
  if (y < 8) { 
    int index;
    if (x % 2 == 0) index = x * 8 + y;
    else index = x * 8 + (7 - y);
    if (index >= 0 && index < 256) ledsOben[index] = color;
  } 
  else { 
    int vX = 31 - x;
    int vY = 7 - (y - 8);
    int index;
    if (vX % 2 == 0) index = vX * 8 + vY;
    else index = vX * 8 + (7 - vY);
    if (index >= 0 && index < 256) ledsUnten[index] = color;
  }
}

void setup() {
  Serial.begin(115200);
  FastLED.addLeds<CHIPSET, LED_PIN_OBEN, COLOR_ORDER>(ledsOben, 256);
  FastLED.addLeds<CHIPSET, LED_PIN_UNTEN, COLOR_ORDER>(ledsUnten, 256);
  FastLED.setBrightness(15);
  FastLED.clear();

  // Startposition der Schlange in der Mitte
  snake[0] = {15, 8};
  snake[1] = {14, 8};
  snake[2] = {13, 8};
}

void loop() {
  // 1. Joystick auslesen & Richtung bestimmen (Schwellenwert 1000/3000)
  int xVal = analogRead(PIN_X);
  int yVal = analogRead(PIN_Y);

  if (xVal < 1000 && dir.x == 0) dir = {-1, 0};      // Links
  else if (xVal > 3000 && dir.x == 0) dir = {1, 0};  // Rechts
  
  // Da Y bei vielen Joysticks invertiert ist, hier die Logik:
  if (yVal < 1000 && dir.y == 0) dir = {0, -1};      // Oben
  else if (yVal > 3000 && dir.y == 0) dir = {0, 1};   // Unten

  // 2. Zeitgesteuerte Bewegung
  if (millis() - lastMoveTime > moveInterval) {
    lastMoveTime = millis();

    // Körper folgt dem Kopf
    for (int i = 2; i > 0; i--) {
      snake[i] = snake[i - 1];
    }

    // Kopf bewegen
    snake[0].x += dir.x;
    snake[0].y += dir.y;

    // --- KEIN VERLIEREN: Teleportation bei Wandberührung ---
    if (snake[0].x >= WIDTH) snake[0].x = 0;
    else if (snake[0].x < 0) snake[0].x = WIDTH - 1;
    
    if (snake[0].y >= HEIGHT_TOTAL) snake[0].y = 0;
    else if (snake[0].y < 0) snake[0].y = HEIGHT_TOTAL - 1;

    // 3. Zeichnen
    FastLED.clear();
    
    // Schlange zeichnen
    for (int i = 0; i < 3; i++) {
      setPixel(snake[i].x, snake[i].y, (i == 0) ? CRGB::White : CRGB::Green);
    }
    
    FastLED.show();
  }
}
