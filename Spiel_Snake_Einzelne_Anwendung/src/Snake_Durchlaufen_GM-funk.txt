#include <Arduino.h>
#include <FastLED.h>

#define LED_PIN_OBEN   26  // Datum oben
#define LED_PIN_UNTEN  25  // Zeit unten
#define COLOR_ORDER    GRB
#define CHIPSET        WS2812B

#define WIDTH  32
#define HEIGHT_TOTAL 16

CRGB ledsOben[256];
CRGB ledsUnten[256];

void setPixel(int x, int y, CRGB color) {
  if (y < 0 || y >= 16 || x < 0 || x >= 32) return;

  if (y < 8) { 
    // OBERES PANEEL (Datum) - Normaler Vertical Zig-Zag
    int index;
    if (x % 2 == 0) { // Gerade Spalte: von oben nach unten
      index = x * 8 + y;
    } else {          // Ungerade Spalte: von unten nach oben
      index = x * 8 + (7 - y);
    }
    if (index >= 0 && index < 256) ledsOben[index] = color;
  } 
  else { 
    // UNTERES PANEEL (Zeit) - Kopfüber montiert
    // Wir müssen x und y für dieses Paneel spiegeln
    int virtualX = 31 - x;
    int virtualY = 7 - (y - 8); // y-8 macht es zu 0-7, 7- macht es kopfüber

    int index;
    // Da das Paneel gedreht ist, ändert sich auch die Zig-Zag Logik
    if (virtualX % 2 == 0) {
      index = virtualX * 8 + virtualY;
    } else {
      index = virtualX * 8 + (7 - virtualY);
    }
    
    if (index >= 0 && index < 256) ledsUnten[index] = color;
  }
}

// --- 2. Die Hilfsfunktion (muss vor loop stehen oder deklariert sein) ---
void showDiagnosticPixel(int x, int y) {
  FastLED.clear();
  
  // Wir zeichnen den aktuellen Pixel weiß
  setPixel(x, y, CRGB::White);
  
  // Farbliche Markierung der Grenze zur Orientierung
  if (y == 7) setPixel(x, y, CRGB::Red);   // Ende des oberen Panels
  if (y == 8) setPixel(x, y, CRGB::Blue);  // Start des unteren Panels

  Serial.printf("Scan: X=%d, Y=%d\n", x, y);
  FastLED.show();
  delay(200); // Sehr langsam (0.2 Sek), um die 3-Pixel-Lücke zu finden
}

void setup() {
  Serial.begin(115200);
  FastLED.addLeds<CHIPSET, LED_PIN_OBEN, COLOR_ORDER>(ledsOben, 256);
  FastLED.addLeds<CHIPSET, LED_PIN_UNTEN, COLOR_ORDER>(ledsUnten, 256);
  FastLED.setBrightness(10);
  FastLED.clear();
}

void loop() {
  // VERTICAL ZIG-ZAG SCAN über das gesamte 32x16 Feld
  for (int x = 0; x < WIDTH; x++) {
    
    // In jeder Spalte X laufen wir die vollen 16 Pixel ab
    if (x % 2 == 0) {
      // Spalte gerade: von ganz oben (0) nach ganz unten (15)
      for (int y = 0; y < HEIGHT_TOTAL; y++) {
        showDiagnosticPixel(x, y);
      }
    } else {
      // Spalte ungerade: von ganz unten (15) nach ganz oben (0)
      for (int y = HEIGHT_TOTAL - 1; y >= 0; y--) {
        showDiagnosticPixel(x, y);
      }
    }
  }
}
