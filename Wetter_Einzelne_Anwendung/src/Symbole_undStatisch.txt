#include <Arduino.h>
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <FastLED.h>
#include <LEDMatrix.h>
#include <LEDText.h>
#include <FontMatrise.h>

// --- Hardware-Konfiguration ---
#define LED_PIN_OBEN   26
#define LED_PIN_UNTEN  25
#define COLOR_ORDER    GRB
#define CHIPSET        WS2812B
#define MATRIX_WIDTH   32
#define MATRIX_HEIGHT  8
#define MATRIX_TYPE    VERTICAL_ZIGZAG_MATRIX

// --- WLAN & API ---
const char* ssid = "Nothin";
const char* password = "nothin099";
const char* apiKey = "343df2364dc5541a3efd274bf2f845df"; 
String stadt = "Wien"; 

// Matrizen
cLEDMatrix<MATRIX_WIDTH, -MATRIX_HEIGHT, MATRIX_TYPE> ledsOben;
cLEDMatrix<-MATRIX_WIDTH, MATRIX_HEIGHT, MATRIX_TYPE> ledsUnten;
cLEDText AnzeigeOben;
cLEDText AnzeigeUnten;

char bufferOben[32];
char bufferUnten[32];
int weatherID = 800; 
float aktuelleTemp = 0;

void setPixel(int x, int y, CRGB color) {
    if (y < 0 || y >= 16 || x < 0 || x >= 32) return;
    
    if (y < 8) { 
        // OBERES PANEEL
        int index = (x % 2 == 0) ? (x * 8 + y) : (x * 8 + (7 - y));
        // ledsOben[0] gibt den CRGB-Pointer zurück, [index] adressiert den Pixel
        ledsOben[0][index] = color; 
    } 
    else { 
        // UNTERES PANEEL (Kopfüber montiert)
        int vX = 31 - x;
        int vY = 7 - (y - 8);
        int index = (vX % 2 == 0) ? (vX * 8 + vY) : (vX * 8 + (7 - y - 8)); // Korrigierte Y-Logik
        ledsUnten[0][index] = color;
    }
}

// --- Icon-Zeichner (Statisch & Animiert) ---
void drawSun() {
    for(int x=25; x<29; x++) for(int y=3; y<7; y++) setPixel(x, y, CRGB::Yellow);
    setPixel(27, 1, CRGB::Yellow); setPixel(27, 8, CRGB::Yellow); // Strahlen oben/unten
}

void drawCloud(int xOff, int yOff, CRGB color) {
    for(int x=xOff; x<xOff+6; x++) setPixel(x, yOff+3, color);
    for(int x=xOff+1; x<xOff+5; x++) setPixel(x, yOff+2, color);
}

void drawRain() {
    drawCloud(24, 1, CRGB::Gray);
    static int animY = 0;
    EVERY_N_MILLISECONDS(200) { animY++; } // Korrektes Makro ohne 'if'
    for(int i=0; i<3; i++) setPixel(25 + (i*2), 5 + ((animY + i) % 3), CRGB::Blue);
}

void drawSnow() {
    drawCloud(24, 1, CRGB::Gray);
    static int animY = 0;
    EVERY_N_MILLISECONDS(400) { animY++; }
    for(int i=0; i<3; i++) setPixel(25 + (i*2), 5 + ((animY + i) % 3), CRGB::White);
}

// --- Wetter-Logik (API v7) ---
void fetchWeather() {
    if (WiFi.status() != WL_CONNECTED) return;
    
    HTTPClient http;
    String url = "http://api.openweathermap.org/data/2.5/weather?q=" + stadt + "&appid=" + String(apiKey) + "&units=metric";
    http.begin(url);
    
    int httpCode = http.GET();
    if (httpCode == 200) {
        JsonDocument doc; // ArduinoJson v7 Standard
        deserializeJson(doc, http.getString());
        aktuelleTemp = doc["main"]["temp"];
        weatherID = doc["weather"][0]["id"];
    }
    http.end();
}

void setup() {
    Serial.begin(115200);

    FastLED.addLeds<CHIPSET, LED_PIN_OBEN, COLOR_ORDER>(ledsOben[0], ledsOben.Size());
    FastLED.addLeds<CHIPSET, LED_PIN_UNTEN, COLOR_ORDER>(ledsUnten[0], ledsUnten.Size());
    FastLED.setBrightness(15);
    FastLED.clear(true);

    AnzeigeOben.SetFont(MatriseFontData);
    AnzeigeOben.Init(&ledsOben, ledsOben.Width(), AnzeigeOben.FontHeight() + 1, 0, 0);
    AnzeigeUnten.SetFont(MatriseFontData);
    AnzeigeUnten.Init(&ledsUnten, ledsUnten.Width(), AnzeigeUnten.FontHeight() + 1, 0, 0);

    AnzeigeOben.SetTextColrOptions(COLR_RGB | COLR_SINGLE, 0x00, 0xff, 0xff); 
    AnzeigeUnten.SetTextColrOptions(COLR_RGB | COLR_SINGLE, 0xff, 0xff, 0xff); 

    WiFi.begin(ssid, password);
    while (WiFi.status() != WL_CONNECTED) { delay(500); }
    
    fetchWeather();
}

void loop() {
    FastLED.clear();

    // Stadt-String vorbereiten (toUpperCase Korrektur)
    String displayStadt = stadt.substring(0, 6);
    displayStadt.toUpperCase(); 
    sprintf(bufferOben, " %s", displayStadt.c_str());
    sprintf(bufferUnten, " %.0f C", aktuelleTemp);

    // Statischen Text setzen (KEIN \x02 nutzen)
    AnzeigeOben.SetText((unsigned char *)bufferOben, strlen(bufferOben));
    AnzeigeUnten.SetText((unsigned char *)bufferUnten, strlen(bufferUnten));

    AnzeigeOben.UpdateText();
    AnzeigeUnten.UpdateText();

    // Icons basierend auf weatherID zeichnen
    if (weatherID == 800) drawSun();
    else if (weatherID >= 801 && weatherID <= 804) drawCloud(24, 2, CRGB::Gray);
    else if (weatherID >= 500 && weatherID <= 531) drawRain();
    else if (weatherID >= 600 && weatherID <= 622) drawSnow();
    else drawCloud(24, 2, CRGB::White);

    FastLED.show();

    // Alle 30 Minuten Wetter aktualisieren
    EVERY_N_MINUTES(30) { fetchWeather(); }
    
    delay(50); 
}
